# parseando strings representando arrays como '[a, b, c]' ou "['a','b']"
def parse_array_string(x):
    if pd.isna(x):
        return []
    s = str(x).strip()
    if s in ["[]", "[[]]", ""]:
        return []
    # remover colchetes externos
    if s.startswith("[") and s.endswith("]"):
        s = s[1:-1]
    # split por v√≠rgula, limpar aspas e espa√ßos
    items = [it.strip().strip("'\"") for it in s.split(",") if it.strip() != ""]
    return items

# Features importantes para os modelos
def get_feature_importances(pipe, feature_names):
    step = pipe.named_steps[list(pipe.named_steps.keys())[-1]]
    if hasattr(step, "feature_importances_"):
        imps = step.feature_importances_
    elif hasattr(step, "coef_"):
        imps = np.abs(step.coef_).ravel()
    else:
        imps = np.zeros(len(feature_names))
    return imps

def multi_class_classification():
    print("=== CONFIGURANDO CLASSIFICA√á√ÉO MULTI-CLASSE ===")
    
    # Usar label2 como target multi-classe
    if 'label2' in df_processed.columns:
        # Mapear labels para n√∫meros
        unique_labels = df_processed['label2'].unique()
        label_mapping = {label: i for i, label in enumerate(unique_labels)}
        reverse_mapping = {i: label for label, i in label_mapping.items()}
        
        print("Mapeamento de classes:")
        for label, num in label_mapping.items():
            print(f"  {label} ‚Üí {num}")
        
        # Criar y multi-classe
        y_multi = df_processed['label2'].map(label_mapping).values
        
        # Selecionar apenas classes com amostras suficientes (ex: > 1000)
        from collections import Counter
        class_counts = Counter(y_multi)
        min_samples = 1000
        valid_classes = [cls for cls, count in class_counts.items() if count >= min_samples]
        
        print(f"\nClasses com amostras suficientes (>={min_samples}): {len(valid_classes)}")
        
        # Filtrar dados para classes v√°lidas
        mask = np.isin(y_multi, valid_classes)
        X_multi = X[mask]
        y_multi_filtered = y_multi[mask]
        
        # Atualizar mapeamento
        valid_mapping = {cls: reverse_mapping[cls] for cls in valid_classes}
        print("Classes finais:", valid_mapping)
        
        if len(valid_classes) > 1:
            return X_multi, y_multi_filtered, valid_mapping
        else:
            print("‚ùå N√£o h√° classes suficientes para classifica√ß√£o multi-classe")
            return None, None, None
    else:
        print("‚ùå Coluna 'label2' n√£o encontrada")
        return None, None, None
